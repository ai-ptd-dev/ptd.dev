# Ruby, Python, or JavaScript
class TodoManager
  def add_task(title, priority: :normal)
    @tasks << Task.new(title, priority)
    save_to_database
  end
end

# Transpiles to Rust, Swift, or C#
impl TodoManager {
    pub fn add_task(&mut self, title: &str, priority: Priority) -> Result<()> {
        let task = Task::new(title, priority);
        self.tasks.push(task);
        self.save_to_database()
    }
}
